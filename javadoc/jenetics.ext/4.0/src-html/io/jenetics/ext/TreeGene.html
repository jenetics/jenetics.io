20 package io.jenetics.ext;
21 
22 import io.jenetics.Gene;
23 import io.jenetics.ext.util.FlatTree;
24 
25 /**
26  * Representation of tree shaped gene. Since the genes are part of a chromosome,
27  * they are implementing the {@link FlatTree} interface, which makes the required
28  * storage layout explicit.
29  *
30  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
31  * @version 3.9
32  * @since 3.9
33  */
34 public interface TreeGene<A, G extends TreeGene<A, G>>
35     extends
36         Gene<A, G>,
37         FlatTree<A, G>
38 {
39 
40     @Override
41     public default A getValue() {
42         return getAllele();
43     }
44 
45     /**
46      * Return a new tree gene with the given allele and the <em>local</em> tree
47      * structure.
48      *
49      * @param allele the actual gene allele
50      * @param childOffset the offset of the first node child within the
51      *        chromosome
52      * @param childCount the number of children of the new tree gene
53      * @return a new tree gene with the given parameters
54      * @throws IllegalArgumentException  if the {@code childCount} is smaller
55      *         than zero
56      */
57     public G newInstance(
58         final A allele,
59         final int childOffset,
60         final int childCount
61     );
62 
63 }
