020 package io.jenetics.ext.grammar;
021 
022 import static java.lang.Character.isWhitespace;
023 import static java.lang.String.format;
024 import static io.jenetics.ext.grammar.Bnf.isIdChar;
025 import static io.jenetics.ext.grammar.Bnf.isStringChar;
026 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.ASSIGN;
027 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.BAR;
028 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.GT;
029 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.ID;
030 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.LT;
031 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.QUOTED_STRING;
032 import static io.jenetics.ext.grammar.BnfTokenizer.BnfTokenType.STRING;
033 
034 import io.jenetics.ext.internal.parser.CharSequenceTokenizer;
035 import io.jenetics.ext.internal.parser.ParsingException;
036 import io.jenetics.ext.internal.parser.Token;
037 
038 /**
039  * Tokenizer for BNF grammars.
040  *
041  * <pre>{@code
042  * ASSIGN: '::=';
043  * BAR: '|';
044  * GT: '>';
045  * LT: '<';
046  * ID: ('a'..'z'|'A'..'Z') ('a'..'z'|'A'..'Z'|'0'..'9'|'-')+;
047  * STRING: ( '%s' | '%i' )? '"' ( ~ '"' )* '"';
048  * WS: [ \r\n\t] -> skip;
049  * }</pre>
050  *
051  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
052  * @since 7.1
053  * @version 7.1
054  */
055 final class BnfTokenizer extends CharSequenceTokenizer {
056 
057     enum BnfTokenType implements Token.Type {
058         ASSIGN(1),
059         BAR(2),
060         GT(3),
061         LT(4),
062         ID(5),
063         STRING(6),
064         QUOTED_STRING(7);
065 
066         private final int _code;
067 
068         BnfTokenType(final int code) {
069             _code = code;
070         }
071 
072         @Override
073         public int code() {
074             return _code;
075         }
076     }
077 
078     BnfTokenizer(final CharSequence input) {
079         super(input);
080     }
081 
082     @Override
083     public Token<String> next() {
084         while (isNonEof(c)) {
085             final char value = c;
086             switch (value) {
087                 case ' ', '\r', '\n', '\t' -> {
088                     WS();
089                 }
090                 case ':' -> {
091                     return ASSIGN();
092                 }
093                 case '|' -> {
094                     consume();
095                     return BAR.token(value);
096                 }
097                 case '>' -> {
098                     consume();
099                     return GT.token(value);
100                 }
101                 case '<' -> {
102                     consume();
103                     return LT.token(value);
104                 }
105                 case '\'' -> {
106                     return QUOTED_STRING();
107                 }
108                 default -> {
109                     if (isAlphabetic(c)) {
110                         return ID();
111                     } else if (!isWhitespace(c)) {
112                         return STRING();
113                     } else {
114                         throw new ParsingException(format(
115                             "Got invalid character '%s' at position '%d'.",
116                             c, pos
117                         ));
118                     }
119                 }
120             }
121         }
122 
123         return null;
124     }
125 
126     private Token<String> ASSIGN() {
127         match(':');
128         match(':');
129         match('=');
130         return ASSIGN.token("::=");
131     }
132 
133     private Token<String> QUOTED_STRING() {
134         final var value = new StringBuilder();
135 
136         match('\'');
137         while (isNonEof(c) && c != '\'') {
138             if (c == '\\') {
139                 consume();
140             }
141 
142             value.append(c);
143             consume();
144         }
145         match('\'');
146 
147         return QUOTED_STRING.token(value.toString());
148     }
149 
150     private Token<String> ID() {
151         final var value = new StringBuilder();
152 
153         while (isIdChar(c)) {
154             value.append(c);
155             consume();
156         }
157 
158         return ID.token(value.toString());
159     }
160 
161     private Token<String> STRING() {
162         final var value = new StringBuilder();
163 
164         while (isNonEof(c) && isStringChar(c)) {
165             value.append(c);
166             consume();
167         }
168 
169         return STRING.token(value.toString());
170     }
