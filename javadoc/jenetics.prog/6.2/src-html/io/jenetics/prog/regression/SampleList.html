020 package io.jenetics.prog.regression;
021 
022 import static java.lang.String.format;
023 
024 import java.io.Serializable;
025 import java.lang.reflect.Array;
026 import java.util.AbstractList;
027 import java.util.List;
028 import java.util.stream.Stream;
029 
030 import io.jenetics.ext.util.Tree;
031 
032 import io.jenetics.prog.op.Op;
033 import io.jenetics.prog.op.Program;
034 
035 /**
036  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
037  * @version 5.0
038  * @since 5.0
039  */
040 final class SampleList<T>
041     extends AbstractList<Sample<T>>
042     implements
043         Sampling<T>,
044         Serializable
045 {
046     private static final long serialVersionUID = 1L;
047 
048     private final List<Sample<T>> _samples;
049 
050     private final Class<T> _type;
051     private final T[][] _arguments;
052     private final T[] _results;
053 
054     @SuppressWarnings("unchecked")
055     SampleList(final List<Sample<T>> samples) {
056         if (samples.isEmpty()) {
057             throw new IllegalArgumentException("Sample list must not be empty.");
058         }
059 
060         _type = (Class<T>)samples.get(0).argAt(0).getClass();
061 
062         final int arity = samples.get(0).arity();
063         if (arity == 0) {
064             throw new IllegalArgumentException(
065                 "The arity of the sample point must not be zero."
066             );
067         }
068 
069         for (int i = 0; i < samples.size(); ++i) {
070             final Sample<T> sample = samples.get(0);
071             if (arity != sample.arity()) {
072                 throw new IllegalArgumentException(format(
073                     "Expected arity %d, but got %d for sample index %d.",
074                     arity, sample.arity(), i
075                 ));
076             }
077         }
078 
079         _samples = samples;
080 
081         _arguments = samples.stream()
082             .map(s -> args(_type, s))
083             .toArray(size -> (T[][])Array.newInstance(_type, size, 0));
084 
085         _results = _samples.stream()
086             .map(Sample::result)
087             .toArray(size -> (T[])Array.newInstance(_type, size));
088     }
089 
090     private static <T> T[] args(final Class<T> type, final Sample<T> sample) {
091         @SuppressWarnings("unchecked")
092         final T[] args = (T[])Array
093             .newInstance(sample.argAt(0).getClass(), sample.arity());
094         for (int i = 0; i < args.length; ++i) {
095             args[i] = sample.argAt(i);
096         }
097 
098         return args;
099     }
100 
101     @Override
102     public Result<T> eval(final Tree<? extends Op<T>, ?> program) {
103         @SuppressWarnings("unchecked")
104         final T[] calculated = Stream.of(_arguments)
105             .map(args -> Program.eval(program, args))
106             .toArray(size -> (T[])Array.newInstance(_type, size));
107 
108         return Result.of(calculated, _results);
109     }
110 
111     @Override
112     public Sample<T> get(int index) {
113         return _samples.get(index);
114     }
115 
116     @Override
117     public int size() {
118         return _samples.size();
119     }
120 
121 }
