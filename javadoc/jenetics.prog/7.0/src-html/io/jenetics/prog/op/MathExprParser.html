038 /**
039  * Simple parser for mathematical expressions.
040  *
041  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>

047     private static final Map<String, Const<Double>> CONST = Map.of(
048         "PI", MathOp.PI,
049         "π", MathOp.PI
050     );

052     /**
053      * Contains the token regex and the token kind;
054      */
055     record TokenDesc(Pattern regex, int type) {}
056 
057     /**
058      * Represents expression token.
059      */
060     record Token(int token, String sequence, int pos) {
061         static final int EPSILON = 0;
062         static final int PLUS = 1;
063         static final int MINUS = 2;
064         static final int MUL = 3;
065         static final int DIV = 4;
066         static final int MOD = 5;
067         static final int POWER = 6;
068         static final int FUNCTION = 7;
069         static final int OPEN_BRACKET = 8;
070         static final int CLOSE_BRACKET = 9;
071         static final int NUMBER = 10;
072         static final int VARIABLE = 11;
073         static final int COMMA = 12;
074 
075         @Override
076         public String toString() {
077             return format("['%s', %s]", sequence, pos);
078         }
079     }

081     /**
082      * The math expression tokenizer.
083      */
084     static final class Tokenizer {
085         static final Tokenizer MATH_OP = new Tokenizer();
086         static {
087             MATH_OP.add("\\+", Token.PLUS);
088             MATH_OP.add("-", Token.MINUS);
089             MATH_OP.add("\\*", Token.MUL);
090             MATH_OP.add("/", Token.DIV);
091             MATH_OP.add("%", Token.MOD);
092             MATH_OP.add("\\^", Token.POWER);
093             MATH_OP.add("\\,", Token.COMMA);
094 
095             final String functions = Stream.of(MathOp.values())
096                 .map(MathOp::toString)
097                 .collect(Collectors.joining("|"));
098             MATH_OP.add("(" + functions + ")(?!\\w)", Token.FUNCTION);
099 
100             MATH_OP.add("\\(", Token.OPEN_BRACKET);
101             MATH_OP.add("\\)", Token.CLOSE_BRACKET);
102             MATH_OP.add("(?:\\d+\\.?|\\.\\d)\\d*(?:[Ee][-+]?\\d+)?", Token.NUMBER);
103             MATH_OP.add("[a-zA-Zπ]\\w*", Token.VARIABLE);
104         }
105 
106         private final Deque<TokenDesc> _infos = new LinkedList<>();
107 
108         private Tokenizer() {
109         }
110 
111         private void add(String regex, int token) {
112             _infos.add(new TokenDesc(Pattern.compile("^(" + regex+")"), token));
113         }
114 
115         Deque<Token> tokenize(final String expression) {
116             final Deque<Token> tokens = new LinkedList<>();
117 
118             String string = expression.trim();
119             final int totalLength = string.length();
120             while (!string.isEmpty()) {
121                 final int remaining = string.length();
122                 boolean match = false;
123                 for (TokenDesc info : _infos) {
124                     final Matcher m = info.regex.matcher(string);
125                     if (m.find()) {
126                         final String tok = m.group().trim();
127                         string = m.replaceFirst("").trim();
128                         tokens.add(new Token(info.type, tok, totalLength - remaining));
129 
130                         match = true;
131                         break;
132                     }
133                 }
134 
135                 if (!match) {
136                     throw new IllegalArgumentException(
137                         "Unexpected character in input: " + string
138                     );
139                 }
140             }
141 
142             return tokens;
143         }
144     }

146     private static final Op<Double> LIST_OP = Const.of(Double.NaN);
147 
148     private final Deque<Token> _tokens;
149     private Token _next;
150 
151     private MathExprParser(final Deque<Token> tokens) {
152         _tokens = requireNonNull(tokens);
153         _next = _tokens.getFirst();
154     }

156     static TreeNode<Op<Double>> parse(final String expr) {
157         if (expr.trim().isEmpty()) {
158             throw new IllegalArgumentException(
159                 "Expression string is empty: " + expr
160             );
161         }
162         return new MathExprParser(Tokenizer.MATH_OP.tokenize(expr)).parse();
163     }
164 
165     private TreeNode<Op<Double>> parse() {
166         final TreeNode<Op<Double>> expr = expression();
167         if (_next.token != Token.EPSILON) {
168             throw new IllegalArgumentException(format(
169                 "Unexpected symbol %s found.", _next
170             ));
171         }
172 
173         Var.reindex(expr);
174         return expr;
175     }
176 
177     private TreeNode<Op<Double>> expression() {
178         return term_10_op_sum(signed_term_10_sum());
179     }
180 
181     private TreeNode<Op<Double>> term_10_op_sum(final TreeNode<Op<Double>> expr) {
182         TreeNode<Op<Double>> result = expr;
183 
184         if (_next.token == Token.PLUS) {
185             final TreeNode<Op<Double>> add = TreeNode
186                 .<Op<Double>>of(MathOp.ADD)
187                 .attach(expr);
188 
189             nextToken();
190             add.attach(term_10_sum());
191             result = term_10_op_sum(add);
192         } else if (_next.token == Token.MINUS) {
193             final TreeNode<Op<Double>> sub = TreeNode
194                 .<Op<Double>>of(MathOp.SUB)
195                 .attach(expr);
196 
197             nextToken();
198             sub.attach(term_10_sum());
199             result = term_10_op_sum(sub);
200         }
201 
202         return result;
203     }
204 
205     private TreeNode<Op<Double>> signed_term_10_sum() {
206         if (_next.token == Token.MINUS) {
207             nextToken();
208             return TreeNode
209                 .<Op<Double>>of(MathOp.NEG)
210                 .attach(term_10_sum());
211         } else if (_next.token == Token.PLUS) {
212             nextToken();
213         }
214 
215         return term_10_sum();
216     }
217 
218     private TreeNode<Op<Double>> term_10_sum() {
219         return term_11_op_mult(term_11_mult());
220     }
221 
222     private TreeNode<Op<Double>> term_11_op_mult(final TreeNode<Op<Double>> expr) {
223         TreeNode<Op<Double>> result = expr;
224 
225         if (_next.token == Token.MUL) {
226             final TreeNode<Op<Double>> prod = TreeNode
227                 .<Op<Double>>of(MathOp.MUL)
228                 .attach(expr);
229 
230             nextToken();
231             prod.attach(signed_term_11_mult());
232             result = term_11_op_mult(prod);
233         } else if (_next.token == Token.DIV) {
234             final TreeNode<Op<Double>> prod = TreeNode
235                 .<Op<Double>>of(MathOp.DIV)
236                 .attach(expr);
237 
238             nextToken();
239             prod.attach(signed_term_11_mult());
240             result = term_11_op_mult(prod);
241         } else if (_next.token == Token.MOD) {
242             final TreeNode<Op<Double>> prod = TreeNode
243                 .<Op<Double>>of(MathOp.MOD)
244                 .attach(expr);
245 
246             nextToken();
247             prod.attach(signed_term_11_mult());
248             result = term_11_op_mult(prod);
249         }
250 
251         return result;
252     }
253 
254     private TreeNode<Op<Double>> signed_term_11_mult() {
255         if (_next.token == Token.MINUS) {
256             nextToken();
257             return TreeNode
258                 .<Op<Double>>of(MathOp.NEG)
259                 .attach(term_11_mult());
260         } else if (_next.token == Token.PLUS) {
261             nextToken();
262         }
263 
264         return term_11_mult();
265     }
266 
267     private TreeNode<Op<Double>> term_11_mult() {
268         return factorOp(argument());
269     }
270 
271     private TreeNode<Op<Double>> factorOp(final TreeNode<Op<Double>> expr) {
272         TreeNode<Op<Double>> result = expr;
273 
274         if (_next.token == Token.POWER) {
275             nextToken();
276 
277             result = TreeNode.<Op<Double>>of(MathOp.POW)
278                 .attach(expr)
279                 .attach(signed_term_11_mult());
280         }
281 
282         return result;
283     }
284 
285     private TreeNode<Op<Double>> argument() {
286         if (_next.token == Token.FUNCTION) {
287             final Op<Double> function = ofName(_next.sequence)
288                 .orElseThrow(() -> new IllegalArgumentException(format(
289                     "Unknown function '%s' found", _next.sequence)));
290 
291             nextToken();
292             final TreeNode<Op<Double>> node = TreeNode.of(function);
293             list(argument(), new ArrayList<>()).forEach(node::attach);
294             return node;
295         } else if (_next.token == Token.COMMA ||
296             _next.token == Token.OPEN_BRACKET)
297         {
298             nextToken();
299             TreeNode<Op<Double>> expr = expression();
300             if (_next.token == Token.COMMA) {
301                 expr = TreeNode
302                     .of(LIST_OP)
303                     .attach(expr)
304                     .attach(argument());
305 
306                 return expr;
307             }
308 
309             if (_next.token != Token.CLOSE_BRACKET) {
310                 throw new IllegalArgumentException(format(
311                     "Closing brackets expected: %s", _next));
312             }
313 
314             nextToken();
315             return expr;
316         }
317 
318         return value();
319     }
320 
321     private static Optional<MathOp> ofName(final String name) {
322         return Stream.of(MathOp.values())
323             .filter(op -> op.name().equalsIgnoreCase(name))
324             .findFirst();
325     }
326 
327     private static List<TreeNode<Op<Double>>> list(
328         final TreeNode<Op<Double>> tree,
329         final List<TreeNode<Op<Double>>> list
330     ) {
331         if (tree.value() == LIST_OP) {
332             tree.childStream().forEach(child -> list(child, list));
333         } else {
334             list.add(tree);
335         }
336         return list;
337     }
338 
339     private TreeNode<Op<Double>> value() {
340         final String value = _next.sequence;
341 
342         if (_next.token == Token.NUMBER) {
343             final TreeNode<Op<Double>> node =
344                 TreeNode.of(Const.of(Double.valueOf(value)));
345 
346             nextToken();
347             return node;
348         }
349 
350         if (_next.token == Token.VARIABLE) {
351             final TreeNode<Op<Double>> node = CONST.containsKey(value)
352                 ? TreeNode.of(CONST.get(value))
353                 : TreeNode.of(Var.of(value, 0));
354 
355             nextToken();
356             return node;
357         }
358 
359         if (_next.token == Token.EPSILON) {
360             throw new IllegalArgumentException("Unexpected end of input.");
361         } else {
362             throw new IllegalArgumentException(format(
363                 "Unexpected symbol %s found.", _next
364             ));
365         }
366     }
367 
368     private void nextToken() {
369         _tokens.pop();
370         _next = _tokens.isEmpty()
371             ? new Token(Token.EPSILON, "", -1)
372             : _tokens.getFirst();
373     }
