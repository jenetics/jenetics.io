021 /**
022  * <h2>Example</h2>
023  *
024  * The following example shows how to solve a GP problem with <em>Jenetics</em>.
025  * We are trying to find a polynomial (or an arbitrary mathematical function)
026  * which approximates a given data set.
027  *
028  * <table summary="Sample points">
029  *     <tr><th>x</th><th>y</th></tr>
030  *     <tr><td>0.00</td><td>0.0000</td></tr>
031  *     <tr><td>0.10</td><td>0.0740</td></tr>
032  *     <tr><td>0.20</td><td>0.1120</td></tr>
033  *     <tr><td>0.30</td><td>0.1380</td></tr>
034  *     <tr><td>...</td><td>...</td></tr>
035  * </table>
036  *
037  * The sample points has been created with the function
038  * <em>f(x) = 4*x^3 - 3*x^2 + x</em>. The knowledge of the creating function
039  * makes it easier to compare the quality of the evolved function. For the
040  * example we created 21 data points.
041  *
042  * <p>
043  *     <b>NOTE</b>: <em>The function which created the sample points is not
044  *     needed in the error function we have to define for the GP. It just let
045  *     us verify the final, evolved result.</em>
046  * </p>
047  *
048  * As first step, we have to define the set of allowed <em>non-terminal</em>
049  * and the <em>terminal</em> operations the GP is working with. Selecting the
050  * right set of operation has a big influence on the performance of the GP. If
051  * the operation set is bigger than necessary, we will expand the potential
052  * search space, and the execution time for finding a solution. For our
053  * <em>polynomial</em> example we will chose the following <em>operations</em>
054  * and <em>terminals</em>.
055  *
056  * <pre>{@code
057  * static final ISeq<Op<Double>> OPERATIONS = ISeq.of(
058  *     MathOp.ADD,
059  *     MathOp.SUB,
060  *     MathOp.MUL
061  * );
062  *
063  * static final ISeq<Op<Double>> TERMINALS = ISeq.of(
064  *     Var.of("x", 0),
065  *     EphemeralConst.of(() -> (double)RandomRegistry.getRandom().nextInt(10))
066  * );
067  * }</pre>
068  *
069  * The chosen non-terminal operation set is sufficient to create any polynomial.
070  * For the terminal operations, we added a variable "x", with index zero, and
071  * an ephemeral integer constant. The purpose of the ephemeral constant is to
072  * create constant values, which will differ for every tree, but stay constant
073  * within a tree.
074  * <p>
075  * In the next step define the fitness function for the GP, which will be an
076  * error function we will minimize.
077  *
078  * <pre>{@code
079  * // The lookup table where the data points are stored.
080  * static final double[][] SAMPLES = new double[][] {
081  *     {-1.0, -8.0000},
082  *     {-0.9, -6.2460},
083  *     ...
084  * };
085  *
086  * static double error(final ProgramGene<Double> program) {
087  *     return Arrays.stream(SAMPLES).mapToDouble(sample -> {
088  *         final double x = sample[0];
089  *         final double y = sample[1];
090  *         final double result = program.eval(x);
091  *         return abs(y - result) + program.size()*0.00005;
092  *     })
093  *     .sum();
094  * }
095  * }</pre>
096  *
097  * The error function calculates the sum of the (absolute) difference between
098  * the sample value and the value calculated the by the evolved <em>program</em>
099  * ({@code ProgramGene}). Since we prefer compact programs over complex one, we
100  * will add a penalty for the program size (the number of nodes of the program
101  * tree).
102  *
103  * <p>
104  *     <b>CAUTION</b>: <em>The penalty for the tree size must be small enough to
105  *     not dominate the error function. We still want to find an approximating
106  *     function and not the smallest possible one</em>
107  * </p>
108  *
109  * After we have defined the error function, we need to define the proper
110  * {@code Codec}.
111  *
112  * <pre>{@code
113  * static final Codec<ProgramGene<Double>, ProgramGene<Double>> CODEC =
114  *     Codec.of(
115  *         Genotype.of(ProgramChromosome.of(
116  *             // Program tree depth.
117  *             5,
118  *             // Chromosome validator.
119  *             ch -> ch.getRoot().size() <= 50,
120  *             OPERATIONS,
121  *             TERMINALS
122  *         )),
123  *         Genotype::getGene
124  *     );
125  * }</pre>
126  *
127  *
128  * There are two particularities in the definition of the
129  * {@code ProgramChromosome}:
130  *
131  * <ol>
132  *     <li>Since we want to narrow the search space, we are limiting the depth
133  *     of newly created program trees to 5.</li>
134  *     <li>Because of crossover operations performed during evolution, the
135  *     resulting programs can grow quite big. To prevent an unlimited growth of
136  *     the program trees, we mark programs with more than _50_ nodes as
137  *     invalid.</li>
138  * </ol>
139  *
140  * Now we are ready to put everything together:
141  *
142  * <pre>{@code
143  * public static void main(final String[] args) {
144  *     final Engine<ProgramGene<Double>, Double> engine = Engine
145  *         .builder(Polynomial::error, CODEC)
146  *         .minimizing()
147  *         .alterers(
148  *             new SingleNodeCrossover<>(),
149  *             new Mutator<>())
150  *         .build();
151  *
152  *     final ProgramGene<Double> program = engine.stream()
153  *         .limit(500)
154  *         .collect(EvolutionResult.toBestGenotype())
155  *         .getGene();
156  *
157  *     System.out.println(Tree.toString(program));
158  * }
159  * }</pre>
160  *
161  * The GP is capable of finding the polynomial which created the sample data.
162  * After a few tries, we got the following (correct) output program:
163  *
164  * <pre>
165  * add
166  * ├── mul
167  * │   ├── x
168  * │   └── sub
169  * │       ├── 0.0
170  * │       └── mul
171  * │           ├── x
172  * │           └── sub
173  * │               ├── sub
174  * │               │   ├── sub
175  * │               │   │   ├── sub
176  * │               │   │   │   ├── 3.0
177  * │               │   │   │   └── x
178  * │               │   │   └── x
179  * │               │   └── x
180  * │               └── x
181  * └── x
182  * </pre>
183  *
184  * This program can be reduced to <em>4*x^3 - 3*x^2 + x</em>, which is exactly
185  * the polynomial, which created the sample data.
186  *
187  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
188  * @version 3.9
189  * @since 3.9
190  */
