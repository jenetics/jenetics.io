20 package io.jenetics.util;
21 
22 import static java.util.Objects.requireNonNull;
23 
24 import java.util.Set;
25 import java.util.function.Supplier;
26 import java.util.stream.Collector;
27 
28 /**
29  * @param <T> the type of input elements to the accumulate operation
30  * @param <A> the accumulator type
31  * @param <R> the result type of the accumulated operation
32  *
33  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
34  * @version 6.1
35  * @since 6.1
36  */
37 final class CollectorAccumulator<T, A, R>
38     implements Accumulator<T, CollectorAccumulator<T, A, R>, R>
39 {
40 
41     private final Collector<T, A, R> _collector;
42     private A _collection;
43 
44     CollectorAccumulator(final Collector<T, A, R> collector) {
45         _collector = requireNonNull(collector);
46     }
47 
48     @Override
49     public CollectorAccumulator<T, A, R>
50     combine(final CollectorAccumulator<T, A, R> other) {
51         if (_collection == null) {
52             _collection = _collector.supplier().get();
53         }
54 
55         _collector.combiner().apply(_collection, other._collection);
56         return this;
57     }
58 
59     @Override
60     public void accept(final T value) {
61         if (_collection == null) {
62             _collection = _collector.supplier().get();
63         }
64         _collector.accumulator().accept(_collection, value);
65     }
66 
67     @Override
68     public R result() {
69         return _collector.finisher().apply(_collection);
70     }
71 
72     @Override
73     public Supplier<CollectorAccumulator<T, A, R>> supplier() {
74         return () -> new CollectorAccumulator<>(_collector);
75     }
76 
77     @Override
78     public Set<Characteristics> characteristics() {
79         return _collector.characteristics();
80     }
81 
82 }
