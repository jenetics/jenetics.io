20 package io.jenetics.engine;
21 
22 import static java.util.Objects.requireNonNull;
23 
24 import java.util.concurrent.CompletableFuture;
25 import java.util.function.Function;
26 
27 import io.jenetics.Gene;
28 import io.jenetics.Genotype;
29 import io.jenetics.Phenotype;
30 import io.jenetics.util.ISeq;
31 import io.jenetics.util.MSeq;
32 import io.jenetics.util.Seq;
33 
34 /**
35  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
36  * @version 5.0
37  * @since 5.0
38  */
39 final class CompletableFutureEvaluator<
40     G extends Gene<?, G>,
41     C extends Comparable<? super C>
42 >
43     implements Evaluator<G, C>
44 {
45 
46     private final Function<? super Genotype<G>, ? extends CompletableFuture<C>> _fitness;
47 
48     CompletableFutureEvaluator(
49         final Function<? super Genotype<G>, ? extends CompletableFuture<C>> fitness
50     ) {
51         _fitness = requireNonNull(fitness);
52     }
53 
54     @Override
55     public ISeq<Phenotype<G, C>> eval(final Seq<Phenotype<G, C>> population) {
56         @SuppressWarnings({"unchecked", "rawtypes"})
57         final CompletableFuture<C>[] evaluate =
58             (CompletableFuture<C>[])population.stream()
59                 .filter(Phenotype::nonEvaluated)
60                 .map(pt -> _fitness.apply(pt.genotype()))
61                 .collect(ISeq.toISeq())
62                 .toArray(new CompletableFuture[0]);
63 
64         final ISeq<Phenotype<G, C>> evaluated = population.stream()
65             .filter(Phenotype::isEvaluated)
66             .collect(ISeq.toISeq());
67 
68         CompletableFuture.allOf(evaluate).join();
69 
70         return evaluated.append(map(population, evaluate));
71     }
72 
73     private ISeq<Phenotype<G, C>> map(
74         final Seq<Phenotype<G, C>> population,
75         final CompletableFuture<C>[] fitnesses
76     ) {
77         final ISeq<Phenotype<G, C>> phenotypes = population.stream()
78             .filter(Phenotype::nonEvaluated)
79             .collect(ISeq.toISeq());
80         assert phenotypes.length() == fitnesses.length;
81 
82         final MSeq<Phenotype<G, C>> result = MSeq.ofLength(phenotypes.size());
83         for (int i = 0; i < fitnesses.length; ++i) {
84             result.set(i, phenotypes.get(i).withFitness(fitnesses[i].join()));
85         }
86 
87         return result.asISeq();
88     }
