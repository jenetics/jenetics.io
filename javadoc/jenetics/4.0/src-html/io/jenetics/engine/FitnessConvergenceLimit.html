021 
022 import static java.lang.Math.min;
023 import static java.lang.String.format;
024 import static java.util.Objects.requireNonNull;
025 
026 import java.util.function.BiPredicate;
027 import java.util.function.DoubleConsumer;
028 import java.util.function.Predicate;
029 import java.util.stream.DoubleStream;
030 import java.util.stream.IntStream;
031 
032 import io.jenetics.stat.DoubleMomentStatistics;
033 import io.jenetics.stat.DoubleMoments;
034 
035 /**
036  * Implementation of the fitness-convergence limit strategy object.
037  *
038  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
039  * @version 3.7
040  * @since 3.7
041  */
042 final class FitnessConvergenceLimit<N extends Number & Comparable<? super N>>
043     implements Predicate<EvolutionResult<?, N>>
044 {
045 
046     private final int _shortFilterSize;
047     private final int _longFilterSize;
048     private final Buffer _buffer;
049     private final BiPredicate<DoubleMoments, DoubleMoments> _proceed;
050 
051     /**
052      * Create a new fitness-convergence limit strategy object.
053      *
054      * @param shortFilterSize the size of the short moments filter
055      * @param longFilterSize the size of the long moments filter
056      * @param proceed the predicate which decides whether to stop or proceed the
057      *        evolution stream
058      * @throws IllegalArgumentException if {@code shortFilterSize < 1} or
059      *         {@code longFilterSize < 2} or
060      *         {@code shortFilterSize >= longFilterSize}
061      * @throws NullPointerException if the {@code proceed} predicate is
062      *         {@code null}
063      */
064     FitnessConvergenceLimit(
065         final int shortFilterSize,
066         final int longFilterSize,
067         final BiPredicate<DoubleMoments, DoubleMoments> proceed
068     ) {
069         if (shortFilterSize < 1) {
070             throw new IllegalArgumentException(format(
071                 "The short filter size must be greater than one: %d",
072                 shortFilterSize
073             ));
074         }
075         if (longFilterSize < 2) {
076             throw new IllegalArgumentException(format(
077                 "The long filter size must be greater than two: %d",
078                 shortFilterSize
079             ));
080         }
081         if (shortFilterSize >= longFilterSize) {
082             throw new IllegalArgumentException(format(
083                 "The long filter size must be greater than the short " +
084                 "filter size: %d <= %d",
085                 longFilterSize, shortFilterSize
086             ));
087         }
088 
089         _shortFilterSize = shortFilterSize;
090         _longFilterSize = longFilterSize;
091         _buffer = new Buffer(longFilterSize);
092         _proceed = requireNonNull(proceed);
093     }
094 
095     @Override
096     public boolean test(final EvolutionResult<?, N> result) {
097         final Number fitness = result.getBestFitness();
098 
099         if (fitness != null) {
100             _buffer.accept(fitness.doubleValue());
101         }
102 
103         return !_buffer.isFull() || _proceed.test(
104             _buffer.doubleMoments(_shortFilterSize),
105             _buffer.doubleMoments(_longFilterSize)
106         );
107     }
108 
109 
110     /**
111      * Ring buffer for {@code double} values. If the buffer is full, old values
112      * are overridden.
113      *
114      * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
115      * @version 3.7
116      * @since 3.7
117      */
118     static final class Buffer implements DoubleConsumer {
119         private final double[] _buffer;
120 
121         private int _pos;
122         private int _length;
123         private long _samples;
124 
125         /**
126          * Create a new double ring-buffer.
127          *
128          * @param capacity the ring buffer capacity.
129          * @throws IllegalArgumentException if {@code capacity < 1}
130          */
131         Buffer(final int capacity) {
132             if (capacity < 1) {
133                 throw new IllegalArgumentException(format(
134                     "Buffer capacity must be greater than one: %d", capacity
135                 ));
136             }
137 
138             _buffer = new double[capacity];
139         }
140 
141         @Override
142         public void accept(final double value) {
143             _buffer[_pos] = value;
144 
145             _pos = (_pos + 1)%_buffer.length;
146             _length = min(_length + 1, _buffer.length);
147             ++_samples;
148         }
149 
150         /**
151          * Return the capacity of the buffer.
152          *
153          * @return the capacity of the buffer
154          */
155         public int capacity() {
156             return _buffer.length;
157         }
158 
159         /**
160          * Return the number of buffer elements.
161          *
162          * @return the number of buffer elements
163          */
164         public int length() {
165             return _length;
166         }
167 
168         /**
169          * Return the overall number of elements the buffer has seen.
170          *
171          * @return the overall number of elements the buffer has seen
172          */
173         public long samples() {
174             return _samples;
175         }
176 
177         /**
178          * Test if the buffer is full.
179          *
180          * @return {@code true} if the buffer is full, {@code false} otherwise
181          */
182         public boolean isFull() {
183             return _length == _buffer.length;
184         }
185 
186         /**
187          * Return a stream of the last ({@code windowSize}) buffer values. The
188          * size if the stream is {@code min(windowSize, length())}.
189          *
190          * @param windowSize the number of stream elements
191          * @return a double value stream
192          */
193         public DoubleStream stream(final int windowSize) {
194             final int length = min(windowSize, _length);
195 
196             return IntStream.range(0, length)
197                 .map(i -> (_pos + _buffer.length - length + i)%_buffer.length)
198                 .mapToDouble(i -> _buffer[i]);
199         }
200 
201         /**
202          * Return the double stream of the buffer values.
203          *
204          * @return the double stream of the buffer values
205          */
206         public DoubleStream stream() {
207             return stream(_length);
208         }
209 
210         /**
211          * Return the double moment statistics of the last {@code windowSize}
212          * buffer values.
213          *
214          * @param windowSize the number of the last double values the statistics
215          *        consists of
216          * @return the double moment statistics of the last {@code windowSize}
217          *         buffer values
218          */
219         public DoubleMoments doubleMoments(final int windowSize) {
220             return DoubleMoments.of(
221                 stream(windowSize).collect(
222                     DoubleMomentStatistics::new,
223                     DoubleMomentStatistics::accept,
224                     DoubleMomentStatistics::combine
225                 )
226             );
227         }
228 
229         /**
230          * Return the double moment statistics of the buffer values.
231          *
232          * @return the double moment statistics of the buffer values
233          */
234         public DoubleMoments doubleMoments() {
235             return doubleMoments(_length);
236         }
237 
238     }
239
