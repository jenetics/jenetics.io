056     /**
057      * Selects a random subset of size {@code k} from a set of size {@code n}.
058      *
059      * @see #subset(int, int[], Random)
060      *
061      * @param n the size of the set.
062      * @param k the size of the subset.
063      * @param random the random number generator used.
064      * @throws NullPointerException if {@code random} is {@code null}.
065      * @throws IllegalArgumentException if {@code n < k}, {@code k == 0} or if
066      *         {@code n*k} will cause an integer overflow.
067      * @return the subset array.
068      */
069     public static int[] subset(final int n, final int k, final Random random) {
070         requireNonNull(random, "Random");
071         if (k <= 0) {
072             throw new IllegalArgumentException(format(
073                 "Subset size smaller or equal zero: %s", k
074             ));
075         }
076         if (n < k) {
077             throw new IllegalArgumentException(format(
078                 "n smaller than k: %s < %s.", n, k
079             ));
080         }
081 
082         final int[] sub = new int[k];
083         subset(n, sub,random);
084         return sub;
085     }

105     /**
106      * <p>
107      * Selects a random subset of size {@code a.length} from a set of size
108      * {@code n}. Implementation of the {@code RANKSB} algorithm described by
109      * <em>Albert Nijenhuis</em> and <em>Herbert Wilf</em> in <b>Combinatorial
110      * Algorithms for Computers and Calculators</b>
111      * </p>
112      *  Reference:</a></em>
113      *      Albert Nijenhuis, Herbert Wilf,
114      *      Combinatorial Algorithms for Computers and Calculators,
115      *      Second Edition,
116      *      Academic Press, 1978,
117      *      ISBN: 0-12-519260-6,
118      *      LC: QA164.N54.
119      *      Page: 42
120      * </p>
121      *
122      * @param n the size of the set.
123      * @param a the a set array.
124      * @param random the random number generator used.
125      * @return the a-set array for the given parameter
126      * @throws NullPointerException if {@code a} or {@code random} is
127      *         {@code null}.
128      * @throws IllegalArgumentException if {@code n < a.length},
129      *         {@code a.length == 0} or {@code n*a.length} will cause an
130      *         integer overflow.
131      */
132     public static int[] subset(final int n, final int a[], final Random random) {
133         requireNonNull(random, "Random");
134         requireNonNull(a, "Sub set array");
135 
136         final int k = a.length;
137         checkSubSet(n, k);
138 
139         // Early return.
140         if (a.length == n) {
141             for (int i = 0; i < k; ++i) a[i] = i;
142             return a;
143         }
144 
145         // (A): Initialize a[i] to "zero" point for bin Ri.
146         for (int i = 0; i < k; ++i) {
147             a[i] = (i*n)/k;
148         }
149 
150         // (B)
151         int l = 0, x = 0;
152         for (int c = 0; c < k; ++c) {
153             do {
154                 // Choose random x;
155                 x = 1 + nextX(random, n - 1);
156 
157                 // determine range Rl;
158                 l = (x*k - 1)/n;
159             } while (a[l] >= x); // accept or reject.
160 
161             ++a[l];
162         }
163         int s = k;
164 
165         // (C) Move a[i] of nonempty bins to the left.
166         int m = 0, p = 0;
167         for (int i = 0; i < k; ++i) {
168             if (a[i] == (i*n)/k) {
169                 a[i] = 0;
170             } else {
171                 ++p;
172                 m = a[i];
173                 a[i] = 0;
174                 a[p - 1] = m;
175             }
176         }
177 
178         // (D) Determine l, set up space for Bl.
179         int ds = 0;
180         for (; p > 0; --p) {
181             l = 1 + (a[p - 1]*k - 1)/n;
182             ds = a[p - 1] - ((l - 1)*n)/k;
183             a[p - 1] = 0;
184             a[s - 1] = l;
185             s -= ds;
186         }
187 
188         // (E) If a[l] != 0, a new bin is to be processed.
189         int r = 0, m0 = 0;
190         for (int ll = 1; ll <= k; ++ll) {
191             l = k + 1 - ll;
192 
193             if (a[l - 1] != 0) {
194                 r = l;
195                 m0 = 1 + ((a[l - 1] - 1)*n)/k;
196                 m = (a[l-1]*n)/k - m0 + 1;
197             }
198 
199             // (F) Choose a random x.
200             x = m0 + nextX(random, m - 1);
201             int i = l + 1;
202 
203             // (G) Check x against previously entered elements in bin;
204             //     increment x as it jumps over elements <= x.
205             while (i <= r && x >= a[i - 1]) {
206                 ++x;
207                 a[i- 2] = a[i - 1];
208                 ++i;
209             }
210 
211             a[i - 2] = x;
212             --m;
213         }
214 
215         return a;
216     }
217 
218     public static void checkSubSet(final int n, final int k) {
219         if (k <= 0) {
220             throw new IllegalArgumentException(format(
221                 "Subset size smaller or equal zero: %s", k
222             ));
223         }
224         if (n < k) {
225             throw new IllegalArgumentException(format(
226                 "n smaller than k: %s < %s.", n, k
227             ));
228         }
229         if (!isMultiplicationSave(n, k)) {
230             throw new IllegalArgumentException(format(
231                 "n*sub.length > Integer.MAX_VALUE (%s*%s = %s > %s)",
232                 n, k, (long)n*(long)k, Integer.MAX_VALUE
233             ));
234         }
235     }
236 
237     private static int nextX(final Random random, final int m) {
238         return m > 0 ? random.nextInt(m) : m - 1;
239     }
