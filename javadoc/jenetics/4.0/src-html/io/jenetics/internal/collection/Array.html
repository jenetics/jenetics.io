036 /**
037  * Array implementation class. This class manages the actual array (store) and
038  * the start index and the length.
039  *
040  * @param <T> the array element type

052     /**

054      *

145     /**
146      * Return a <i>new</i> {@code Array} object with the given values appended.
147      *
148      * @since 3.4
149      *
150      * @param array the values to append
151      * @return a <i>new</i> {@code Array} object with the elements of
152      *         {@code this} array and the given {@code array} appended.
153      * @throws NullPointerException if the given {@code array} is {@code null}
154      */
155     public Array<T> append(final Array<T> array) {
156         final Array<T> appended = newInstance(length() + array.length());
157         for (int i = 0; i < _length; ++i) {
158             appended.set(i, get(i));
159         }
160         for (int i = 0; i < array._length; ++i) {
161             appended.set(i + _length, array.get(i));
162         }
163 
164         return appended;
165     }

171     /**
172      * Return a <i>new</i> {@code Array} with the given {@code values} appended.
173      *
174      * @since 3.4
175      *
176      * @param values the values to append
177      * @return a <i>new</i> {@code Array} with the elements of {@code this}
178      *        array and the given {@code values} appended.
179      * @throws NullPointerException if the given {@code values} iterable is
180      *         {@code null}
181      */
182     public Array<T> append(final Iterable<? extends T> values) {
183         final int size = size(values);
184         final Array<T> array = newInstance(_length + size);
185 
186         for (int i = 0; i < _length; ++i) {
187             array.set(i, get(i));
188         }
189 
190         final Iterator<? extends T> it = values.iterator();
191         for (int i = 0; i < size; ++i) {
192             array.set(_length + i, it.next());
193         }
194 
195         return array;
196     }

198     /**
199      * Return a <i>new</i> {@code Array} with the given {@code values} prepended.
200      *
201      * @since 3.4
202      *
203      * @param values the values to prepend
204      * @return a <i>new</i> {@code Array} with the elements of {@code this}
205      *        array and the given {@code values} appended.
206      * @throws NullPointerException if the given {@code values} iterable is
207      *         {@code null}
208      */
209     public Array<T> prepend(final Iterable<? extends T> values) {
210         final int size = size(values);
211         final Array<T> array = newInstance(_length + size);
212 
213         final Iterator<? extends T> it = values.iterator();
214         for (int i = 0; i < size; ++i) {
215             array.set(i, it.next());
216         }
217 
218         for (int i = 0; i < _length; ++i) {
219             array.set(size + i, get(i));
220         }
221 
222         return array;
223     }

225     private static int size(final Iterable<?> values) {
226         int size = 0;
227         if (values instanceof Collection<?>) {
228             size = ((Collection<?>)values).size();
229         } else {
230             for (Object value : values) {
231                 ++size;
232             }
233         }
234 
235         return size;
236     }

281     /**
282      * Check the given array {@code index}
283      *
284      * @param index the index to check
285      * @throws ArrayIndexOutOfBoundsException if the given index is not in the
286      *         valid range.

288     public void checkIndex(final int index) {
289         if (index < 0 || index >= length()) {
290             throw new ArrayIndexOutOfBoundsException(format(
291                 "Index %s is out of bounds [0, %s)", index, length()
292             ));
293         }
294     }

296     /**
297      * Check the given {@code from} and {@code until} indices.
298      *
299      * @param from the from index, inclusively.
300      * @param until the until index, exclusively.
301      * @throws ArrayIndexOutOfBoundsException if the given index is not in the
302      *         valid range.
303      */
304     public final void checkIndex(final int from, final int until) {
305         if (from > until) {
306             throw new ArrayIndexOutOfBoundsException(format(
307                 "fromIndex(%d) > toIndex(%d)", from, until
308             ));
309         }
310         if (from < 0 || until > length()) {
311             throw new ArrayIndexOutOfBoundsException(format(
312                 "Invalid index range: [%d, %s)", from, until
313             ));
314         }
315     }

340     private void writeObject(final ObjectOutputStream out)
341         throws IOException
342     {
343         out.defaultWriteObject();
344 
345         final Store<T> store = _start == 0
346             ? _store._value
347             : _store._value.copy(_start, _start + _length);
348 
349         out.writeBoolean(_store._sealed);
350         out.writeObject(store);
351     }

353     private void readObject(final ObjectInputStream in)
354         throws IOException, ClassNotFoundException
355     {
356         in.defaultReadObject();
357 
358         final boolean sealed = in.readBoolean();
359         @SuppressWarnings("unchecked")
360         final Store<T> store = (Store<T>) in.readObject();
361         final Store.Ref<T> ref = new Ref<>(store, sealed);
362 
363         reflect.setField(this, "_start", 0);
364         reflect.setField(this, "_length", store.length());
365         reflect.setField(this, "_store", ref);
366     }
