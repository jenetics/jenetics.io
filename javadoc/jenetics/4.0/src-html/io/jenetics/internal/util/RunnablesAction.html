20 package io.jenetics.internal.util;
21 
22 import static java.lang.Math.max;
23 import static java.security.AccessController.doPrivileged;
24 
25 import java.security.PrivilegedAction;
26 import java.util.concurrent.RecursiveAction;
27 
28 import io.jenetics.util.Seq;
29 
30 /**
31  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
32  * @version 4.0
33  * @since 2.0
34  */
35 final class RunnablesAction extends RecursiveAction {
36     private static final long serialVersionUID = 1;
37 
38     private static final int DEFAULT_THRESHOLD = 7;
39 
40     private final Seq<? extends Runnable> _runnables;
41     private final int _high;
42     private final int _low;
43 
44     private RunnablesAction(
45         final Seq<? extends Runnable> runnables,
46         final int low,
47         final int high
48     ) {
49         _runnables = runnables;
50         _low = low;
51         _high = high;
52     }
53 
54     public RunnablesAction(final Seq<? extends Runnable> runnables) {
55         this(runnables, 0, runnables.size());
56     }
57 
58     @Override
59     protected void compute() {
60         if ((_high - _low) <= Env.splitThreshold ||
61             getSurplusQueuedTaskCount() > Env.maxSurplusQueuedTaskCount)
62         {
63             for (int i = _low; i < _high; ++i) {
64                 _runnables.get(i).run();
65             }
66         } else {
67             final int mid = (_low + _high) >>> 1;
68             invokeAll(
69                 new RunnablesAction(_runnables, _low, mid),
70                 new RunnablesAction(_runnables, mid, _high)
71             );
72         }
73     }
74 
75     private static final class Env {
76         private static final int splitThreshold = max(
77             doPrivileged(
78                 (PrivilegedAction<Integer>)() -> Integer.getInteger(
79                     "io.jenetics.concurrency.splitThreshold",
80                     5
81                 )),
82             1
83         );
84 
85         private static final int maxSurplusQueuedTaskCount = max(
86             doPrivileged(
87                 (PrivilegedAction<Integer>)() -> Integer.getInteger(
88                     "io.jenetics.concurrency.maxSurplusQueuedTaskCount",
89                     3
90                 )),
91             1
92         );
93     }
