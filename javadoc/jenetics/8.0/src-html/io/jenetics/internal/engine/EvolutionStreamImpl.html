27 
28 import io.jenetics.Gene;
29 import io.jenetics.engine.Evolution;
30 import io.jenetics.engine.EvolutionResult;

32 import io.jenetics.engine.EvolutionStream;

36 /**
37  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
38  * @since 3.0
39  * @version 5.1
40  */
41 public final class EvolutionStreamImpl<
42     G extends Gene<?, G>,
43     C extends Comparable<? super C>
44 >
45     extends StreamProxy<EvolutionResult<G, C>>
46     implements EvolutionStream<G, C>
47 {
48 
49     private final Spliterator<EvolutionResult<G, C>> _spliterator;
50 
51     public EvolutionStreamImpl(
52         final Spliterator<EvolutionResult<G, C>> spliterator,
53         final boolean parallel
54     ) {
55         super(StreamSupport.stream(spliterator, parallel));
56         _spliterator = spliterator;
57     }
58 
59     public EvolutionStreamImpl(
60         final Supplier<EvolutionStart<G, C>> start,
61         final Evolution<G, C> evolution
62     ) {
63         this(new EvolutionSpliterator<>(start, evolution), false);
64     }
65 
66     @Override
67     public EvolutionStream<G, C>
68     limit(final Predicate<? super EvolutionResult<G, C>> proceed) {
69         return new EvolutionStreamImpl<>(
70             LimitSpliterator.of(_spliterator, proceed),
71             isParallel()
72         );
73     }
74 
75     public static <G extends Gene<?, G>, C extends Comparable<? super C>>
76     EvolutionStreamImpl<G, C> of(
77         final Supplier<EvolutionStart<G, C>> start,
78         final Function<
79             ? super EvolutionStart<G, C>,
80             ? extends Evolution<G, C>> evolution
81     ) {
82         return new EvolutionStreamImpl<>(
83             EvolutionSpliterator.of(start, evolution),
84             false
85         );
86     }
