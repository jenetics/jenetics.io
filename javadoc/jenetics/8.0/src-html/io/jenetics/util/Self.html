20 package io.jenetics.util;

22 /**
23  * This interface defines a <em>recursive</em> generic type {@code S}, which
24  * represents the type of the implementing class.
25  * {@snippet lang="java":
26  * interface Foo<T extends Foo<T>> extends Self<T> {
27  *     // ...
28  * }
29  * }
30  * Using the {@code Self} interface in this case makes it clear that the generic
31  * type {@code T} of the interface {@code Foo} represents the concrete type of
32  * the class, implementing the interface {@code Foo}.
33  * <p>
34  * If the interface is used as intended, the following generic {@code min} method
35  * can be implemented as a <em>default</em> method.
36  * {@snippet lang="java":
37  * interface Foo<A extends Foo<A>> extends Self<A>, Comparable<A> {
38  *     // ...
39  *
40  *     default A max(final A other) {
41  *         return compareTo(other) > 0 ? self() : other;
42  *     }
43  * }
44  * }
45  *
46  * @param <S> the type of the implementing class.
47  *
48  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
49  * @since 7.0
50  * @version 7.0
51  */
52 public interface Self<S extends Self<S>> {
53 
54     /**
55      * Return a reference of {@code this} object as the declared generic type
56      * {@code S}.
57      *
58      * @return the {@code this} reference as the generic type {@code S}
59      * @throws ClassCastException if the interface is not used as intended and
60      *         {@code (this instanceof S) == false}
61      */
62     @SuppressWarnings("unchecked")
63     default S self() {
64         return (S)this;
65     }
66 
67 }
