353     static final HeapSorter INSTANCE = new HeapSorter();
354 
355     @Override
356     public <T> int[] sort(
357         final Seq<? extends T> array,
358         final int[] indexes,
359         final Comparator<? super T> comparator
360     ) {
361         // Heapify
362         for (int k = array.size()/2; k >= 0; --k) {
363             sink(array, indexes, comparator, k, array.size());
364         }
365 
366         // Sort down.
367         for (int i = array.size(); --i >= 1;) {
368             swap(indexes, 0, i);
369             sink(array, indexes, comparator, 0, i);
370         }
371 
372         return indexes;
373     }

375     private static <T> void sink(
376         final Seq<? extends T> array,
377         final int[] indexes,
378         final Comparator<? super T> comparator,
379         final int start,
380         final int end
381     ) {
382         int m = start;
383         while (2*m < end) {
384             int j = 2*m;
385             if (j < end - 1 && comparator.compare(array.get(indexes[j]), array.get(indexes[j + 1])) > 0) ++j;
386             if (comparator.compare(array.get(indexes[m]), array.get(indexes[j])) <= 0) break;
387             swap(indexes, m, j);
388             m = j;
389         }
390     }

449     private static void sink(
450         final int[] array,
451         final int[] indexes,
452         final IntComparator comparator,
453         final int start,
454         final int end
455     ) {
456         int m = start;
457         while (2*m < end) {
458             int j = 2*m;
459             if (j < end - 1 &&
460                 comparator.compare(array[indexes[j]], array[indexes[j + 1]]) > 0)
461             {
462                 ++j;
463             }
464             if (comparator.compare(array[indexes[m]], array[indexes[j]]) <= 0) {
