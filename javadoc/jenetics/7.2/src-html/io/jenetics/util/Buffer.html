023 import static java.lang.System.arraycopy;
024

026 import java.util.Iterator;
027 import java.util.function.IntFunction;

029 import io.jenetics.internal.collection.Array;

033 /**
034  * This class is a bounded buffer, which can store only a given amount of
035  * elements. If the buffer is full, it starts <em>overwriting</em> previously
036  * inserted elements at the beginning of the buffer. A <em>full</em> buffer
037  * neither blocks the insertion of new elements, nor does it throw an exception.
038  * <p>
039  * If you want to access the elements of the buffer, you have to take a
040  * <em>snapshot</em> of the current content. This can be done with the
041  * {@link #toArray()}, {@link #toArray(IntFunction)} and {@link #toSeq()} methods.
042  *
043  * @implNote
044  * This class is not thread-safe. If two threads access the buffer
045  * concurrently it must be <em>synchronized</em> externally.
046  *
047  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
048  * @version 5.1
049  * @since 5.0
050  */
051 final class Buffer<T> implements Iterable<T> {
052     private final Object[] _buffer;
053 
054     private int _index;
055     private int _size;
056 
057     /**
058      * Create a new ring buffer with the given {@code capacity}.
059      *
060      * @param capacity the buffer capacity
061      * @throws NegativeArraySizeException if the given {@code capacity} is
062      *         negative
063      */
064     private Buffer(final int capacity) {
065         _buffer = new Object[capacity];
066     }
067 
068     /**
069      * Return the capacity of {@code this} buffer. This is the number of elements
070      * the buffer is able to store.
071      *
072      * @return the capacity of {@code this} buffer
073      */
074     public int capacity() {
075         return _buffer.length;
076     }
077 
078     /**
079      * Return the current number of elements {@code this} buffer holds.
080      *
081      * @return the current buffer size
082      */
083     public int size() {
084         return _size;
085     }
086 
087     /**
088      * Returns {@code true} if this buffer contains no elements.
089      *
090      * @return {@code true} if this buffer contains no elements
091      */
092     public boolean isEmpty() {
093         return _size == 0;
094     }
095 
096     /**
097      * Add a new element to the buffer.
098      *
099      * @param value the value to add
100      */
101     public void add(final T value) {
102         _buffer[_index] = value;
103 
104         if (++_index == _buffer.length) {
105             _index = 0;
106         }
107         if (_size < _buffer.length) {
108             ++_size;
109         }
110     }
111 
112     /**
113      * Add the given {@code values} to the buffer.
114      *
115      * @param values the values to add
116      * @throws NullPointerException if the given parameter is {@code null}
117      */
118     @SafeVarargs
119     public final void addAll(final T... values) {
120         addAll(values, 0, values.length);
121     }
122 
123     /**
124      * Add the given values to the ring buffer.
125      *
126      * @param values the values being added to the ring buffer
127      * @throws NullPointerException if the given parameter is {@code null}
128      */
129     public void addAll(final Iterable<? extends T> values) {
130         if (values instanceof Buffer<?> buff) {
131             final Object[] array = buff.toArray();
132             addAll(array, 0, array.length);
133         } else if (values instanceof Collection<?> coll) {
134             final Object[] array = coll.toArray();
135             addAll(array, 0, array.length);
136         } else {
137             for (T value : values) {
138                 add(value);
139             }
140         }
141     }
142 
143     /**
144      * Add the {@code values} of the given array to {@code this} buffer. The
145      * values are added at the given {@code start} index and the given
146      * {@code length}.
147      *
148      * @param values the array which contains the values to add
149      * @param start the start index of the source array
150      * @param length the number of elements to copy
151      * @throws IndexOutOfBoundsException if copying causes access of data
152      *         outside array bounds
153      * @throws ArrayStoreException if an element in the {@code value} array
154      *         could not be stored into the dest array because of a type mismatch
155      */
156     private void addAll(final Object[] values, final int start, final int length) {
157         if (length >= _buffer.length) {
158             arraycopy(
159                 values, values.length - _buffer.length + start,
160                 _buffer, 0, _buffer.length
161             );
162             _size = _buffer.length;
163             _index = 0;
164         } else {
165             final int remaining = _buffer.length - _index;
166             _size = min(_size + length, _buffer.length);
167 
168             if (length <= remaining) {
169                 arraycopy(values, start, _buffer, _index, length);
170                 _index += length;
171             } else {
172                 arraycopy(values, start, _buffer, _index, remaining);
173                 arraycopy(
174                     values, remaining + start,
175                     _buffer, 0, length - remaining
176                 );
177                 _index = length - remaining;
178             }
179         }
180     }
181 
182     /**
183      * Return an iterator from an element snapshot, taken with the {@link #toSeq()}
184      * method.
185      *
186      * @return a new iterator from a snapshot of the current elements
187      */
188     @Override
189     public Iterator<T> iterator() {
190         return toSeq().iterator();
191     }
192 
193     /**
194      * Return a snapshot of the current buffer content.
195      *
196      * @return the buffer snapshot
197      */
198     public Object[] toArray() {
199         return toArray(Object[]::new);
200     }
201 
202     /**
203      * Return a snapshot of the current buffer content.
204      *
205      * @param generator a function which produces a new array of the desired
206      *        type and the provided length
207      * @param <A> the element type of the resulting array
208      * @return the buffer snapshot
209      * @throws ArrayStoreException if an element in the {@code value} array
210      *         could not be stored into the dest array because of a type mismatch
211      * @throws NullPointerException if the given {@code generator} is {@code null}
212      */
213     public <A> A[] toArray(final IntFunction<A[]> generator) {
214         final A[] result = generator.apply(_size);
215         copyTo(result);
216         return result;
217     }
218 
219     private <A> void copyTo(final A[] array) {
220         assert array.length >= _size;
221 
222         if (_size < _buffer.length || _index == 0) {
223             arraycopy(_buffer, 0, array, 0, _size);
224         } else {
225             arraycopy(_buffer, _index, array, 0, _buffer.length - _index);
226             arraycopy(_buffer, 0, array, _buffer.length - _index, _index);
227         }
228     }
229 
230     /**
231      * Return a snapshot of the current buffer content.
232      *
233      * @return the buffer snapshot
234      */
235     public ISeq<T> toSeq() {
236         return isEmpty()
237             ? ISeq.empty()
238             : new ArrayMSeq<T>(Array.of(ObjectStore.of(toArray()))).toISeq();
239     }
240 
241     @Override
242     public String toString() {
243         return toSeq().toString();
244     }
245 
246     /**
247      * Create a new ring buffer with the given {@code capacity}.
248      *
249      * @param <T> the element type
250      * @param capacity the buffer capacity
251      * @return a new ring buffer with the given capacity
252      * @throws NegativeArraySizeException if the given {@code capacity} is
253      *         negative
254      */
255     public static <T> Buffer<T> ofCapacity(final int capacity) {
256         return new Buffer<>(capacity);
257     }
