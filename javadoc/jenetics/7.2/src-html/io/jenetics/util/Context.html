020 package io.jenetics.util;
021 
022 import java.util.concurrent.atomic.AtomicReference;
023 import java.util.function.Function;
024 
025 /**
026  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
027  * @version 3.0
028  * @since 2.0
029  */
030 final class Context<T> {
031 
032     private final T _default;
033     private final AtomicReference<Entry<T>> _entry;
034     private final ThreadLocal<Entry<T>> _threadLocalEntry = new ThreadLocal<>();
035 
036     Context(final T defaultValue) {
037         _default = defaultValue;
038         _entry = new AtomicReference<>(new Entry<>(defaultValue));
039     }
040 
041     void set(final T value) {
042         final Entry<T> e = _threadLocalEntry.get();
043 
044         if (e != null) e.value = value;
045         else _entry.set(new Entry<>(value));
046     }
047 
048     T get() {
049         final Entry<T> e = _threadLocalEntry.get();
050         return (e != null ? e : _entry.get()).value;
051     }
052 
053     void reset() {
054         set(_default);
055     }
056 
057     <S extends T, R> R with(final S value, final Function<S, R> f) {
058         final Entry<T> e = _threadLocalEntry.get();
059         if (e != null) {
060             _threadLocalEntry.set(e.inner(value));
061         } else {
062             _threadLocalEntry.set(new Entry<>(value, Thread.currentThread()));
063         }
064 
065         try {
066             return f.apply(value);
067         } finally {
068             _threadLocalEntry.set(_threadLocalEntry.get().parent);
069         }
070     }
071 
072     /**
073      * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
074      * @version 2.0
075      * @since 2.0
076      */
077     private static final class Entry<T> {
078         final Thread thread;
079         final Entry<T> parent;
080         T value;
081 
082         Entry(final T value, final Entry<T> parent, final Thread thread) {
083             this.value = value;
084             this.parent = parent;
085             this.thread = thread;
086         }
087 
088         Entry(final T value, final Thread thread) {
089             this(value, null, thread);
090         }
091 
092         Entry(final T value) {
093             this(value, null, null);
094         }
095 
096         Entry<T> inner(final T value) {
097             assert thread == Thread.currentThread();
098             return new Entry<>(value, this, thread);
099         }
100 
101     }
102 
103 }
