20 package io.jenetics;
21 
22 import io.jenetics.util.Mean;
23 
24 /**
25  * Alters a chromosome by replacing two genes by its mean value.
26  *
27  * <p>
28  * The order ({@link #order()}) of this recombination implementation is two.
29  * </p>
30  *
31  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
32  * @since 1.0
33  * @version 6.0
34  */
35 public class MeanAlterer<
36     G extends Gene<?, G> & Mean<G>,
37     C extends Comparable<? super C>
38 >
39     extends CombineAlterer<G, C>
40 {
41 
42     /**
43      * Constructs an alterer with a given recombination probability.
44      *
45      * @param probability the crossover probability.
46      * @throws IllegalArgumentException if the {@code probability} is not in the
47      *         valid range of {@code [0, 1]}.
48      */
49     public MeanAlterer(final double probability) {
50         super(Mean::mean, probability);
51     }
52 
53     /**
54      * Create a new alterer with alter probability of {@code 0.05}.
55      */
56     public MeanAlterer() {
57         this(0.05);
58     }
59 
60 }
