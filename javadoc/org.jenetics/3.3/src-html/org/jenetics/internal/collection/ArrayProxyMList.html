22 /**
23  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>

25  * @version 1.4
26  */
27 public class ArrayProxyMList<T, P extends ArrayProxy<T, ?, ?>>
28     extends ArrayProxyList<T, P>
29 {
30     private static final long serialVersionUID = 1L;
31 
32     public ArrayProxyMList(final P proxy) {
33         super(proxy);
34     }
35 
36     @Override
37     public T set(final int index, final T element) {
38         proxy.cloneIfSealed();
39         final T oldElement = proxy.get(index);
40         proxy.set(index, element);
41         return oldElement;
42     }
43 
44 }
