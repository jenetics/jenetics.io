20 package org.jenetics.internal.collection;
21 
22 import static java.util.Objects.requireNonNull;
23 
24 import java.util.Spliterator;
25 import java.util.function.Consumer;
26 
27 import org.jenetics.util.Seq;
28 
29 /**
30  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
31  * @since 3.0
32  * @version 3.0
33  */
34 public class SeqSpliterator<T> implements Spliterator<T> {
35 
36     private final Seq<T> _seq;
37     private final int _fence;
38     private int _index;
39 
40     public SeqSpliterator(
41         final Seq<T> seq,
42         final int origin,
43         final int fence
44     ) {
45         _seq = requireNonNull(seq);
46         _index = origin;
47         _fence = fence;
48     }
49 
50     public SeqSpliterator(final Seq<T> seq) {
51         this(seq, 0, seq.length());
52     }
53 
54     @Override
55     public void forEachRemaining(final Consumer<? super T> action) {
56         requireNonNull(action);
57 
58         Seq<T> seq;
59         int i;
60         int hi;
61 
62         if ((seq = _seq).length() >= (hi = _fence) &&
63             (i = _index) >= 0 && i < (_index = hi))
64         {
65             do {
66                 action.accept(seq.get(i));
67             } while (++i < hi);
68         }
69     }
70 
71     @Override
72     public boolean tryAdvance(final Consumer<? super T> action) {
73         if (_index >= 0 && _index < _fence) {
74             action.accept(_seq.get(_index++));
75             return true;
76         }
77         return false;
78     }
79 
80     @Override
81     public Spliterator<T> trySplit() {
82         final int lo = _index;
83         final int mid = (lo + _fence) >>> 1;
84 
85         return (lo >= mid)
86             ? null
87             : new SeqSpliterator<>(_seq, lo, _index = mid);
88     }
89 
90     @Override
91     public long estimateSize() {
92         return _fence - _index;
93     }
94 
95     @Override
96     public int characteristics() {
97         return Spliterator.NONNULL | Spliterator.SIZED | Spliterator.SUBSIZED;
98     }
99 }
