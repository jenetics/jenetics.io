035 /**
036  * Abstraction for an ordered and bounded sequence of elements.
037  *
038  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
039  * @since 1.4
040  * @version 3.1
041  */
042 public abstract class ArrayProxy<T, A, P extends ArrayProxy<T, A, P>>
043     implements
044         Copyable<P>,
045         Serializable
046 {
047     private static final long serialVersionUID = 1L;
048 
049     /**
050      * This flag determines if {@code this} represents a sealed proxy instance.
051      */
052     transient boolean _isSealedProxy = false;
053 
054     /**
055      * Contains all sealed proxies, which share the same {@code array} as
056      * {@code this} proxy instance.
057      */
058     transient Stack<WeakReference<ArrayProxy<?, ?, ?>>>
059         _sealedProxies = new Stack<>();
060 
061     /**
062      * Used for creating new array proxy instances.
063      */
064     private final ArrayProxyFactory<A, P> _proxyFactory;
065 
066     /**
067      * Used for creating copies of the current array.
068      */
069     private final ArrayCopier<A> _arrayCopier;

071     /**
072      * The actual array, where the elements are stored.
073      */
074     public A array;

079     public int start;

082      * The end index of the array view, exclusively.

084     public int end;

086     /**
087      * The actual array length.
088      */
089     public final int length;

093     /**
094      * Create a new array proxy.
095      *
096      * @param array the array which is wrapped by this proxy
097      * @param start the start index of the wrapped array (inclusively)
098      * @param end the end index of the wrapped array (exclusively)
099      * @param factory factory function for creating new proxy objects
100      * @param copier array cloning function
101      * @throws java.lang.NullPointerException if one of the arguments is
102      *         {@code null}
103      * @throws java.lang.IllegalArgumentException if the start and end indexes
104      *         are invalid.
105      */
106     protected ArrayProxy(
107         final A array,
108         final int start,
109         final int end,
110         final ArrayProxyFactory<A, P> factory,
111         final ArrayCopier<A> copier
112     ) {
113         if (start < 0 || end < 0 || end < start) {
114             throw new IllegalArgumentException(format(
115                 "Invalid indexes [%d, %d)", start, end
116             ));
117         }
118 
119         this.array = requireNonNull(array);
120         this.length = end - start;
121         this.start = start;
122         this.end = end;
123 
124         _proxyFactory = requireNonNull(factory);
125         _arrayCopier = requireNonNull(copier);
126     }

128     /**
129      * Return the <i>array</i> element at the specified, absolute position in
130      * the {@code ArrayProxy}. The array boundaries are not checked.
131      *
132      * @param index absolute index of the element to return
133      * @return the <i>array</i> element at the specified absolute position
134      */
135     public abstract T __get__(final int index);
136 
137     /**
138      * Set the <i>array</i> element at the specified absolute position in the
139      * {@code ArrayProxy}. The array boundaries are not checked.
140      *
141      * @param index absolute index of the <i>array</i> element
142      * @param value the value to set
143      */
144     public abstract void __set__(final int index, final T value);
145 
146     /**
147      * Return the <i>array</i> element at the specified position in the
148      * {@code ArrayProxy}. The array boundaries are not checked.
149      *
150      * @param index index of the element to return
151      * @return the <i>array</i> element at the specified position
152      */
153     public final T __get(final int index) {
154         return __get__(index + start);
155     }
156 
157     /**
158      * Set the <i>array</i> element at the specified position in the
159      * {@code ArrayProxy}. The array boundaries are not checked.
160      *
161      * @param index index of the <i>array</i> element
162      * @param value the value to set
163      */
164     public final void __set(final int index, final T value) {
165         __set__(index + start, value);
166     }
167 
168     /**
169      * Return the <i>array</i> element at the specified position in the
170      * {@code ArrayProxy}.
171      *
172      * @param index index of the element to return
173      * @return the <i>array</i> element at the specified position
174      * @throws IndexOutOfBoundsException if the index it out of range
175      *         (index &lt; 0 || index &gt;= _length).
176      */
177     public final T get(final int index) {
178         checkIndex(index);
179         return __get__(index + start);
180     }

182     /**
183      * Set the <i>array</i> element at the specified position in the
184      * {@code ArrayProxy}
185      *
186      * @param index the index of the element to set
187      * @param value the <i>array</i> element
188      * @throws IndexOutOfBoundsException if the index it out of range
189      *         {@code (index &lt; 0 || index &gt;= _length)}.
190      */
191     public final void set(final int index, final T value) {
192         checkIndex(index);
193         __set__(index + start, value);
194     }

196     /**
197      * Return a new sub {@code ArrayProxy} object with the given start and end
198      * indexes. The underlying array storage is not copied. With the returned
199      * sub-array proxy it is possible to <i>write through</i> the original
200      * array.
201      *
202      * @param from the start index of the new sub {@code ArrayProxy} object,
203      *        inclusively.
204      * @param until the end index of the new sub {@code ArrayProxy} object,
205      *        exclusively.
206      * @return a new array proxy (view) with the given start and end index.
207      * @throws IndexOutOfBoundsException if the given indexes are out of bounds.
208      */
209     public final P slice(final int from, final int until) {
210         final P slice = _proxyFactory.create(array, from + start, until + start);
211         slice._isSealedProxy = _isSealedProxy;
212         slice._sealedProxies = _sealedProxies;
213 
214         if (_isSealedProxy) {
215             _sealedProxies.push(new WeakReference<>(slice));
216         }
217 
218         return slice;
219     }
220 
221     /**
222      * Return a new sub {@code ArrayProxy} object with the given start index.
223      * The underlying array storage is not copied. With the returned sub-array
224      * proxy it is possible to <i>write through</i> the original array.
225      *
226      * @param from the start index of the new sub {@code ArrayProxy} object,
227      *        inclusively.
228      * @return a new array proxy (view) with the given start index.
229      * @throws IndexOutOfBoundsException if the given indexes are out of bounds.
230      */
231     public final P slice(final int from) {
232         return slice(from, length);
233     }
234 
235     /**
236      * Set the seal flag for this {@code ArrayProxy} instance and return a new
237      * {@code ArrayProxy} object with an not set <i>seal</i> flag but with the
238      * same underlying data structure.
239      *
240      * @return a new {@code ArrayProxy} instance; for command chaining.
241      */
242     public final P seal() {
243         assert !_isSealedProxy : "Must not be called on sealed proxies";
244 
245         final P proxy = _proxyFactory.create(array, start, end);
246         proxy._sealedProxies = _sealedProxies;
247         proxy._isSealedProxy = true;
248         _sealedProxies.push(new WeakReference<>(proxy));
249 
250         return proxy;
251     }
252 
253     /**
254      * Clone the underlying data structure of this {@code ArrayProxy} if it is
255      * sealed.
256      */
257     public final void cloneIfSealed() {
258         assert !_isSealedProxy : "Must not be called on sealed proxies";
259 
260         if (_sealedProxies.length > 0) {
261             _sealedProxies.popAll(p -> {
262                 final ArrayProxy<?, ?, ?> proxy = p.get();
263                 if (proxy != null) {
264                     proxy.copyArray();
265                 }
266             });
267         }
268     }
269 
270     private void copyArray() {
271         assert _isSealedProxy : "Must only be called on sealed proxies";
272 
273         array = _arrayCopier.copy(array, start, end);
274         _sealedProxies = new Stack<>();
275         _isSealedProxy = false;
276         start = 0;
277         end = length;
278     }

306     /**
307      * Swap a given range with a range of the same size with another array.
308      * Implementations of this class should replace this with a optimized
309      * version, depending on the underlying data structure.
310      *
311      * <pre>
312      *            from                until
313      *              |                   |
314      * this:  +---+---+---+---+---+---+---+---+---+---+---+---+
315      *              +---------------+
316      *                          +---------------+
317      * other: +---+---+---+---+---+---+---+---+---+---+---+---+
318      *                          |
319      *                      otherFrom
320      * </pre>
321      *
322      * @param start the start index of {@code this} range, inclusively.
323      * @param end the end index of {@code this} range, exclusively.
324      * @param other the other array to swap the elements with.
325      * @param otherStart the start index of the {@code other} array.
326      * @throws IndexOutOfBoundsException if {@code start > end} or
327      *         if {@code from < 0 || until >= this.length() || otherFrom < 0 ||
328      *         otherFrom + (until - from) >= other.length()}
329      */
330     public void swap(
331         final int start,
332         final int end,
333         final ArrayProxy<T, ?, ?> other,
334         final int otherStart
335     ) {
336         checkIndex(start, end);
337         other.checkIndex(otherStart, otherStart + (end - start));
338         cloneIfSealed();
339         other.cloneIfSealed();
340 
341         for (int i = end - start; --i >= 0;) {
342             final T temp = __get(i + start);
343             __set(i + start, other.__get(otherStart + i));
344             other.__set(otherStart + i, temp);
345         }
346     }

348     public final <B> ObjectArrayProxy<B> map(
349         final Function<? super T, ? extends B> mapper
350     ) {
351         return map(mapper, ObjectArrayProxy<B>::new);
352     }

354     public final <B, P extends ArrayProxy<B, ?, ?>> P map(
355         final Function<? super T, ? extends B> mapper,
356         final IntFunction<P> builder
357     ) {
358         final P result = builder.apply(length);
359         assert result.length == length;
360 
361         for (int i = 0; i < length; ++i) {
362             result.__set__(i, mapper.apply(__get(i)));
363         }
364         return result;
365     }

373     /**
374      * Checks the given index.s
375      *
376      * @param start the index to check.
377      * @throws java.lang.ArrayIndexOutOfBoundsException if the given index is
378      *         not in the valid range.
379      */
380     protected final void checkIndex(final int start) {
381         if (start < 0 || start >= length) {
382             throw new ArrayIndexOutOfBoundsException(format(
383                 "Index %s is out of bounds [0, %s)", start, length
384             ));
385         }
386     }
387 
388     /**
389      * Check the given {@code from} and {@code until} indices.
390      *
391      * @param start the start index, inclusively.
392      * @param end the end index, exclusively.
393      * @throws java.lang.ArrayIndexOutOfBoundsException if the given index is
394      *         not in the valid range.
395      */
396     protected final void checkIndex(final int start, final int end) {
397         if (start > end) {
398             throw new ArrayIndexOutOfBoundsException(format(
399                 "fromIndex(%d) > toIndex(%d)", start, end
400             ));
401         }
402         if (start < 0 || end > length) {
403             throw new ArrayIndexOutOfBoundsException(format(
404                 "Invalid index range: [%d, %s)", start, end
405             ));
406         }
