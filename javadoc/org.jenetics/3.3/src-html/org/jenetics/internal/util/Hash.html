024 /**
025  * Interface for calculating the object hash value.
026  *
027  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
028  * @since 1.6

030  */
031 public interface Hash {
032 
033     /**

037      * @return {@code this}
