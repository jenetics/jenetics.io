046     private static final class JAXBContextHolder {
047         private static final JAXBContext CONTEXT; static {
048             try {
049                 CONTEXT = JAXBContext.newInstance(
050                     "org.jenetics:org.jenetics.internal.util"
051                 );
052             } catch (JAXBException e) {
053                 throw new DataBindingException(
054                     "Something went wrong while creating JAXBContext.", e
055                 );
056             }
057         }
058     }
059 
060     public static JAXBContext context() {
061         return JAXBContextHolder.CONTEXT;
062     }

064     private static final XmlAdapter<Object, Object> IDENTITY_ADAPTER =
065         new XmlAdapter<Object, Object>() {
066             @Override public Object unmarshal(final Object value) {
067                 return value;
068             }
069             @Override public Object marshal(final Object value) {
070                 return value;
071             }
072         };
073 
074     private static final ConcurrentMap<Class<?>, XmlAdapter<?, ?>> ADAPTERS =
075         new ConcurrentHashMap<>();
076 
077     static {
078         ADAPTERS.put(Character.class, CharacterModel.ADAPTER);
079         ADAPTERS.put(CharacterModel.class, CharacterModel.ADAPTER);
080     }
081 
082     /**
083      * Return the an {@code XmlAdapter} for the given {@code vale}. If no
084      * adapter could be found, and identity adapter is returned.
085      *
086      * @param value the object for which to find an {@code XmlAdapter}
087      * @return the {@code XmlAdapter} for the given object, or an identity
088      *         adapter if no one can be found.
089      */
090     @SuppressWarnings("unchecked")
091     public static XmlAdapter<Object, Object> adapterFor(final Object value) {
092         return (XmlAdapter<Object, Object>)ADAPTERS
093             .computeIfAbsent(classOf(value), jaxb::newXmlAdapter);
094     }
095 
096     private static XmlAdapter<Object, Object> newXmlAdapter(final Class<?> cls) {
097         return innerClasses(cls)
098             .filter(XmlAdapter.class::isAssignableFrom)
099             .findFirst()
100             .flatMap(reflect::<XmlAdapter<Object, Object>>newInstance)
101             .orElse(IDENTITY_ADAPTER);
102     }
103 
104     /**
105      * Shorthand for {@code adapterFor(value).marshal(value)}
106      */
107     public static Object marshal(final Object value) throws Exception {
108         return adapterFor(value).marshal(value);
109     }
110 
111     /**
112      * Shorthand for {@code adapterFor(value).unmarshal(value)}
113      */
114     public static Object unmarshal(final Object value) throws Exception {
115         return adapterFor(value).unmarshal(value);
116     }
117 
118     /**
119      * Return a marshaller function from the given
120      * {@link javax.xml.bind.annotation.adapters.XmlAdapter}.
121      *
122      * @param a the adapter used by the marshaller function.
123      * @return the marshaller function
124      */
125     public static <V, B> Function<B, V> Marshaller(final XmlAdapter<V, B> a) {
126         return value -> {
127             try {
128                 return a.marshal(value);
129             } catch (Exception e) {
130                 throw new RuntimeException(e);
131             }
132         };
133     }
134 
135     /**
136      * Return a unmarshaller function from the given
137      * {@link javax.xml.bind.annotation.adapters.XmlAdapter}.
138      *
139      * @param a the adapter used by the unmarshaller function.
140      * @return the unmarshaller function
141      */
142     public static <V, B> Function<V, B> Unmarshaller(final XmlAdapter<V, B> a) {
143         return value -> {
144             try {
145                 return a.unmarshal(value);
146             } catch (Exception e) {
147                 throw new RuntimeException(e);
148             }
149         };
150     }
151 
152     /**
153      * Return a marshaller function for the given object.
154      *
155      * @param value the value to marshal
156      * @return the marshaller function
157      */
158     public static Function<Object, Object> Marshaller(final Object value) {
159         return Marshaller(adapterFor(value));
160     }
161 
162     /**
163      * Return a unmarshaller function for the given object.
164      *
165      * @param value the value to unmarshal
166      * @return the unmarshaller function
167      */
168     public static Function<Object, Object> Unmarshaller(final Object value)  {
169         return Unmarshaller(jaxb.adapterFor(value));
170     }
