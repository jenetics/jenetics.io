091     /**
092      * Tests whether a predicate holds for all elements of this sequence.
093      *
094      * @param predicate the predicate to use to test the elements.
095      * @return {@code true} if the given predicate p holds for all elements of
096      *         this sequence, {@code false} otherwise.
097      * @throws NullPointerException if the given {@code predicate} is
098      *         {@code null}.
099      */
100     public default boolean forAll(final Predicate<? super T> predicate) {
101         boolean valid = true;
102 
103         if (this instanceof RandomAccess) {
104             for (int i = 0, n = length(); i < n && valid; ++i) {
105                 valid = predicate.test(get(i));
106             }
107         } else {
108             final Iterator<T> it = iterator();
109             while (it.hasNext() && valid) {
110                 valid = predicate.test(it.next());
111             }
112         }
113 
114         return valid;
115     }

456      * order; the runtime type of the returned array is that of the specified
457      * array. If this sequence fits in the specified array, it is returned
458      * therein. Otherwise, a new array is allocated with the runtime type of the
459      * specified array and the length of this array.
460      * <p>
461      * If this sequence fits in the specified array with room to spare (i.e.,
462      * the array has more elements than this array), the element in the array
463      * immediately following the end of this array is set to null. (This is
464      * useful in determining the length of the array only if the caller knows
465      * that the list does not contain any null elements.)
466      *
467      * @see java.util.Collection#toArray(Object[])
468      *
469      * @param array the array into which the elements of this array are to be
470      *         stored, if it is big enough; otherwise, a new array of the same
471      *         runtime type is allocated for this purpose.
472      * @return an array containing the elements of this array
473      * @throws ArrayStoreException if the runtime type of the specified array is
474      *         not a super type of the runtime type of every element in this
475      *         array
476      * @throws NullPointerException if the given array is {@code null}.
477      */
478     @SuppressWarnings("unchecked")
479     public default T[] toArray(final T[] array) {
480         if (array.length < length()) {
481             final T[] copy = (T[])java.lang.reflect.Array.newInstance(
482                 array.getClass().getComponentType(), length()
483             );
484             for (int i = length(); --i >= 0;) {
485                 copy[i] = get(i);
486             }
487 
488             return copy;
489         }
490 
491         for (int i = 0, n = length(); i < n; ++i) {
492             array[i] = get(i);
493         }
494         return array;
495     }

649      * final Iterator<E> it = seq.iterator();

670     /**
671      * Unified method for compare to sequences for equality.
672      *
673      * @see Seq#equals(Object)
674      *
675      * @param seq the sequence to test for equality.
676      * @param obj the object to test for equality with the sequence.
677      * @return {@code true} if the given objects are sequences and contain the
678      *          same objects in the same order, {@code false} otherwise.
679      */
680     public static boolean equals(final Seq<?> seq, final Object obj) {
681         if (obj == seq) {
682             return true;
683         }
684         if (!(obj instanceof Seq<?>)) {
685             return false;
686         }
687 
688         final Seq<?> other = (Seq<?>)obj;
689         boolean equals = seq.length() == other.length();
690         for (int i = seq.length(); equals && --i >= 0;) {
691             final Object element = seq.get(i);
692             if (element != null) {
693                 equals = element.equals(other.get(i));
694             } else {
695                 equals = other.get(i) == null;
696             }
697         }
698         return equals;
699     }
