034 /**
035  * Immutable, ordered, fixed sized sequence.
036  *
037  * @see MSeq
038  *
039  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
040  * @since 1.0
041  * @version 3.3
042  */
043 public interface ISeq<T>
044     extends
045         Seq<T>,
046         Copyable<MSeq<T>>
047 {

057 
058     /**
059      * Return a shallow copy of this sequence. The sequence elements are not
060      * cloned.
061      *
062      * @return a shallow copy of this sequence.
063      */
064     @Override
065     public MSeq<T> copy();

072     /**
073      * Single instance of an empty {@code ISeq}.
074      *
075      * @since 3.3
076      */
077     public static final ISeq<?> EMPTY =
078         new ArrayProxyMSeq<>(new ObjectArrayProxy<>(0)).toISeq();
079 
080     /**
081      * Return an empty {@code ISeq}.
082      *
083      * @since 3.3
084      *
085      * @param <T> the element type of the returned {@code ISeq}.
086      * @return an empty {@code ISeq}.
087      */
088     @SuppressWarnings("unchecked")
089     public static <T> ISeq<T> empty() {
090         return (ISeq<T>)EMPTY;
091     }
092 
093     /**
094      * Returns a {@code Collector} that accumulates the input elements into a
095      * new {@code ISeq}.
096      *
097      * @param <T> the type of the input elements
098      * @return a {@code Collector} which collects all the input elements into an
099      *         {@code ISeq}, in encounter order
100      */
101     public static <T> Collector<T, ?, ISeq<T>> toISeq() {
102         return Collector.of(
103             (Supplier<List<T>>)ArrayList::new,
104             List::add,
105             (left, right) -> { left.addAll(right); return left; },
106             ISeq::of
107         );
108     }
109 
110     /**
111      * Create a new {@code ISeq} from the given values.
112      *
113      * @param <T> the element type
114      * @param values the array values.
115      * @return a new {@code ISeq} with the given values.
116      * @throws NullPointerException if the {@code values} array is {@code null}.
117      */
118     @SafeVarargs
119     public static <T> ISeq<T> of(final T... values) {
120         return values.length == 0
121             ? empty()
122             : MSeq.of(values).toISeq();
123     }
124 
125     /**
126      * Create a new {@code ISeq} from the given values.
127      *
128      * @param <T> the element type
129      * @param values the array values.
130      * @return a new {@code ISeq} with the given values.
131      * @throws NullPointerException if the {@code values} array is {@code null}.
132      */
133     @SuppressWarnings("unchecked")
134     public static <T> ISeq<T> of(final Iterable<? extends T> values) {
135         requireNonNull(values);
136 
137         return values instanceof ISeq<?>
138             ? (ISeq<T>)values
139             : values instanceof MSeq<?>
140                 ? ((MSeq<T>)values).toISeq()
141                 : MSeq.of(values).toISeq();
142     }

160     /**
161      * Creates a new sequence, which is filled with objects created be the given
162      * {@code supplier}.
163      *
164      * @since 3.2
165      *
166      * @param <T> the element type of the sequence
167      * @param supplier the {@code Supplier} which creates the elements, the
168      *        returned sequence is filled with
169      * @param length the length of the returned sequence
170      * @return a new sequence filled with elements given by the {@code supplier}
171      * @throws NegativeArraySizeException if the given {@code length} is
172      *         negative
173      * @throws NullPointerException if the given {@code supplier} is
174      *         {@code null}
175      */
176     public static <T> ISeq<T> of(
177         final Supplier<? extends T> supplier,
178         final int length
179     ) {
180         requireNonNull(supplier);
181         require.nonNegative(length);
182 
183         return length == 0
184             ? empty()
185             : MSeq.<T>ofLength(length).fill(supplier).toISeq();
186     }
