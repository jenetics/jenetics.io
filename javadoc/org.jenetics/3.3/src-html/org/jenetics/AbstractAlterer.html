20 package org.jenetics;
21 
22 import static org.jenetics.internal.util.Equality.eq;
23 
24 import org.jenetics.internal.util.Equality;
25 import org.jenetics.internal.util.Hash;
26 import org.jenetics.internal.util.require;
27 
28 /**
29  * Abstract implementation of the alterer interface.
30  *
31  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
32  * @since 1.0
33  * @version 3.0
34  */
35 public abstract class AbstractAlterer<
36     G extends Gene<?, G>,
37     C extends Comparable<? super C>
38 >
39     implements Alterer<G, C>
40 {
41 
42     /**
43      * The altering probability.
44      */
45     protected final double _probability;
46 
47     /**
48      * Constructs an alterer with a given recombination probability.
49      *
50      * @param probability The recombination probability.
51      * @throws IllegalArgumentException if the {@code probability} is not in the
52      *         valid range of {@code [0, 1]}.
53      */
54     protected AbstractAlterer(final double probability) {
55         _probability = require.probability(probability);
56     }
57 
58     /**
59      * Return the recombination/alter probability for this alterer.
60      *
61      * @return The recombination probability.
62      */
63     public double getProbability() {
64         return _probability;
65     }
66 
67     @Override
68     public int hashCode() {
69         return Hash.of(getClass()).and(_probability).value();
70     }
71 
72     @Override
73     public boolean equals(final Object obj) {
74         return Equality.of(this, obj).test(alterer ->
75             eq(_probability, alterer._probability)
76         );
77     }
78 }
