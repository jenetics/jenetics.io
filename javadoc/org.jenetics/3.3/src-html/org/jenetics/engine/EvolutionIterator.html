20 package org.jenetics.engine;
21 
22 import static java.util.Objects.requireNonNull;
23 
24 import java.util.Iterator;
25 import java.util.function.Function;
26 import java.util.function.Supplier;
27 
28 import org.jenetics.Gene;
29 
30 /**
31  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
32  * @since 3.0
33  * @version 3.0
34  */
35 final class EvolutionIterator<
36     G extends Gene<?, G>,
37     C extends Comparable<? super C>
38 >
39     implements Iterator<EvolutionResult<G, C>>
40 {
41 
42     private final Function<EvolutionStart<G, C>, EvolutionResult<G, C>> _evolution;
43     private final Supplier<EvolutionStart<G, C>> _initial;
44 
45     private EvolutionStart<G, C> _start;
46 
47     EvolutionIterator(
48         final Function<EvolutionStart<G, C>, EvolutionResult<G, C>> evolution,
49         final Supplier<EvolutionStart<G, C>> initial
50     ) {
51         _evolution = requireNonNull(evolution);
52         _initial = requireNonNull(initial);
53     }
54 
55     @Override
56     public EvolutionResult<G, C> next() {
57         if (_start == null) {
58             _start = _initial.get();
59         }
60 
61         final EvolutionResult<G, C> result = _evolution.apply(_start);
62         _start = result.next();
63         return result;
64     }
65 
66     @Override
67     public boolean hasNext() {
68         return true;
69     }
70 
71 }
