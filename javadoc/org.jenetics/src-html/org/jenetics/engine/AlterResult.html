20 package org.jenetics.engine;
21 
22 import static java.util.Objects.requireNonNull;
23 
24 import org.jenetics.Gene;
25 import org.jenetics.Population;
26 
27 /**
28  * Represents the result of the alter step.
29  *
30  * @param <G> the gene type
31  * @param <C> the fitness type
32  *
33  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
34  * @since 3.0
35  * @version 3.0
36  */
37 final class AlterResult<
38     G extends Gene<?, G>,
39     C extends Comparable<? super C>
40 > {
41 
42     final Population<G, C> population;
43     final int alterCount;
44 
45     AlterResult(
46         final Population<G, C> population,
47         final int alterCount
48     ) {
49         this.population = requireNonNull(population);
50         this.alterCount = alterCount;
51     }
52 
53 }
