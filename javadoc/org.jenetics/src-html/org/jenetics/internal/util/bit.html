179         final byte[] data, final int start, final int end,
180         final byte[] otherData, final int otherStart
181     ) {
182         for (int i = end - start; --i >= 0;) {
183             final boolean temp = get(data, i + start);
184             set(data, i + start, get(otherData, otherStart + i));
185             set(otherData, otherStart + i, temp);
186         }
187     }

214     /**
215      * Shifting all bits in the given {@code data} array the given
216      * {@code shift} to the right. The bits on the left side are filled with
217      * zeros.
218      *
219      * @param data the data bits to shift.
220      * @param shift the number of bits to shift.
221      * @return the given {@code data} array.
222      * @throws NullPointerException if the {@code data} array is {@code null}.
223      */
224     public static byte[] shiftRight(final byte[] data, final int shift) {
225         final int bytes = min(shift >>> 3, data.length);
226         final int bits = shift & 7;
227 
228         if (bytes > 0) {
229             for (int i = 0, n = data.length - bytes; i < n; ++i) {
230                 data[i] = data[i + bytes];
231             }
232             for (int i = data.length, n = data.length - bytes; --i >= n;) {
233                 data[i] = (byte)0;
234             }
235         }
236         if (bits > 0 && bytes < data.length) {
237             int carry = 0;
238             int nextCarry = 0;
239 
240             for (int i = data.length; --i >= 0;) {
241                 int d = data[i] & 0xFF;
242                 nextCarry = d << (Byte.SIZE - bits);
243 
244                 d >>>= bits;
245                 d |= carry;
246                 data[i] = (byte)(d & 0xFF);
247 
248                 carry = nextCarry;
249             }
250         }
251 
252         return data;
253     }

394         final int to = min(data.length << 3, end);
395         final int byteStart = start >>> 3;
396         final int bitStart = start & 7;
397         final int bitLength = to - start;
398 
399         final byte[] copy = new byte[toByteLength(to - start)];
400 
401         if (copy.length > 0) {
402             // Perform the byte wise right shift.
403             System.arraycopy(data, byteStart, copy, 0, copy.length);
404 
405             // Do the remaining bit wise right shift.
406             shiftRight(copy, bitStart);
407 
408             // Add the 'lost' bits from the next byte, if available.
409             if (data.length > copy.length + byteStart) {
410                 copy[copy.length - 1] |= (byte)(data[byteStart + copy.length]
411                     << (Byte.SIZE - bitStart));
412             }
413 
414             // Trim (delete) the overhanging bits.
415             copy[copy.length - 1] &= 0xFF >>> ((copy.length << 3) - bitLength);
416         }
417 
418         return copy;
419     }

429      * string has the following format:
430      * <pre>
431      *  Byte:       3        2        1        0
432      *              |        |        |        |
433      *  Array: "11110011|10011101|01000000|00101010"
434      *          |                 |        |      |
435      *  Bit:    23                15       7      0
436      * </pre>
437      * <i>Only the array string is printed.</i>
438      *
439      * @see #fromByteString(String)
440      *
441      * @param data the byte array to convert to a string.
442      * @return the binary representation of the given byte array.
443      */
444     public static String toByteString(final byte... data) {
445         final StringBuilder out = new StringBuilder();
446 
447         if (data.length > 0) {
448             for (int j = 7; j >= 0; --j) {
449                 out.append((data[data.length - 1] >>> j) & 1);
450             }
451         }
452         for (int i = data.length - 2; i >= 0 ;--i) {
453             out.append('|');
454             for (int j = 7; j >= 0; --j) {
455                 out.append((data[i] >>> j) & 1);
456             }
457         }
458 
459         return out.toString();
460     }
