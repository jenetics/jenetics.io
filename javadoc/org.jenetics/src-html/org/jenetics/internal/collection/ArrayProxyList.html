020 package org.jenetics.internal.collection;
021 
022 import static java.util.Objects.requireNonNull;
023 
024 import java.io.Serializable;
025 import java.lang.reflect.Array;
026 import java.util.AbstractList;
027 import java.util.RandomAccess;
028 
029 /**
030  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
031  * @since 1.4
032  * @version 3.0
033  */
034 public class ArrayProxyList<T, P extends ArrayProxy<T, ?, ?>>
035     extends AbstractList<T>
036     implements
037         RandomAccess,
038         Serializable
039 {
040     private static final long serialVersionUID = 1L;
041 
042     public final P proxy;
043 
044     public ArrayProxyList(final P proxy) {
045         this.proxy = requireNonNull(proxy, "ArrayProxy must not be null.");
046     }
047 
048     @Override
049     public T get(final int index) {
050         return proxy.get(index);
051     }
052 
053     @Override
054     public int size() {
055         return proxy.length;
056     }
057 
058     @Override
059     public int indexOf(final Object element) {
060         int index = -1;
061         if (element == null) {
062             for (int i = proxy.start; i < proxy.end && index == -1; ++i) {
063                 if (proxy.__get__(i) == null) {
064                     index = i - proxy.start;
065                 }
066             }
067         } else {
068             for (int i = proxy.start; i < proxy.end && index == -1; ++i) {
069                 if (element.equals(proxy.__get__(i))) {
070                     index = i - proxy.start;
071                 }
072             }
073         }
074 
075         return index;
076     }
077 
078     @Override
079     public int lastIndexOf(final Object element) {
080         int index = -1;
081         if (element == null) {
082             for (int i = proxy.end; --i >= proxy.start && index == -1;) {
083                 if (proxy.__get__(i) == null) {
084                     index = i - proxy.start;
085                 }
086             }
087         } else {
088             for (int i = proxy.end; --i >= proxy.start && index == -1;) {
089                 if (element.equals(proxy.__get__(i))) {
090                     index = i - proxy.start;
091                 }
092             }
093         }
094 
095         return index;
096     }
097 
098     @Override
099     public boolean contains(final Object element) {
100         return indexOf(element) != -1;
101     }
102 
103     @Override
104     public Object[] toArray() {
105         final Object[] array = new Object[size()];
106         for (int i = size(); --i >= 0;) {
107             array[i] = proxy.__get(i);
108         }
109         return array;
110     }
111 
112     @SuppressWarnings("unchecked")
113     @Override
114     public <E> E[] toArray(final E[] array) {
115         if (array.length < size()) {
116             final E[] copy = (E[])Array.newInstance(
117                 array.getClass().getComponentType(), size()
118             );
119             for (int i = size(); --i >= 0;) {
120                 copy[i] = (E) proxy.__get(i);
121             }
122 
123             return copy;
124         }
125 
126         for (int i = size(); --i >= 0;) {
127             array[i] = (E) proxy.__get(i);
128         }
129         return array;
130     }
